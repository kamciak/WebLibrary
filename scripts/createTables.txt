create table "APP".WL_USER
(
	USRLOGIN VARCHAR(11) not null primary key,
	USRPASSWORD VARCHAR(50) not null,
	USRNAME VARCHAR(50),
	USRSURNAME VARCHAR(50),
	USRCOUNTRY VARCHAR(50),
	USRCITY VARCHAR(50),
	USRSTREET VARCHAR(50),
	USRPOSTCODE VARCHAR(50),
	USRPHONE VARCHAR(50),
	USRISADMIN BOOLEAN
);

create table "APP".BOOK
(
	ID BIGINT not null primary key GENERATED ALWAYS AS IDENTITY
	(START WITH 1, INCREMENT BY 1),
	TITLE VARCHAR(100),
	AUTHOR VARCHAR(100),
	ISBN VARCHAR(100),
	BOOKYEAR INTEGER,
	AVAILABLE BOOLEAN
);

create table "APP".RESERVATION
(
	USRID VARCHAR(11) not null,
	BOOKID BIGINT not null,
	DATE DATE,
	primary key (USRID, BOOKID)
);

create table "APP".BORROW
(
	ID BIGINT not null primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	USRID VARCHAR(11) not null,
	BOOKID BIGINT not null,
	DATE DATE not null,
	DELETED BOOLEAN not null,
    RETURNDATE DATE
);
